       ```typescript

       import { createX402PaymentHook } from './hooks/x402-payment';
       ```

    2. Add x402 server initialization. Place this after the existing imports and before the `const server = Fastify(...)` line. Use lazy initialization pattern (the x402 server is created but initialize() is called in the start() function since it's async):
       ```typescript
       // x402 Resource Server (lazy init -- initialize() called in start())
       // Import conditionally to avoid hard failure if packages not installed
       let x402Server: any = null;
       async function initX402() {
         try {
           const { x402ResourceServer } = await import('@x402/core/server');
           const { ExactEvmScheme } = await import('@x402/evm/exact/server');
           const { HTTPFacilitatorClient } = await import('@x402/core/server');

           const facilitatorUrl = process.env.X402_FACILITATOR_URL || 'https://x402.org/facilitator';
           const facilitatorClient = new HTTPFacilitatorClient({ url: facilitatorUrl });
           x402Server = new x402ResourceServer(facilitatorClient)
             .register(process.env.X402_NETWORK || 'eip155:84532', new ExactEvmScheme());
           await x402Server.initialize();
           server.log.info('x402 payment system initialized');
         } catch (err) {
           server.log.warn({ err }, 'x402 payment system not available (packages missing or facilitator unreachable)');
         }
       }
       ```

       NOTE: If dynamic import paths fail (e.g., '@x402/core/server' not found), try '@x402/core' and access server exports from there. Check actual package exports after Plan 01 install. The executor should verify the import paths work before committing.

    3. Register x402 preHandler BEFORE the Unkey auth hook. Move the existing Unkey hook registration and insert x402 before it. The new ordering in index.ts should be:

       ```typescript
       // Free-tier rate limiting (100 calls/day by IP, skips authenticated requests)
       registerFreeTierRateLimit(server);

       // x402 machine payment verification (preHandler, BEFORE Unkey auth)
       // If valid x402 payment header present, sets billingPath='x402' and skips auth
       server.addHook('preHandler', async (request, reply) => {
         if (x402Server && request.url.startsWith('/api/v1/')) {
           const walletAddress = process.env.RECEIVING_WALLET_ADDRESS;
           if (walletAddress) {
             const handler = createX402PaymentHook(
               x402Server,
               walletAddress,
               process.env.X402_PRICE_PER_CALL || '$0.001',
               process.env.X402_NETWORK || 'eip155:84532',
             );
             await handler(request, reply);
           }
         }
       });

       // Unkey auth middleware for public API routes (/api/v1/*)
       // MODIFIED: Skip if billingPath already set by x402 payment hook
       const unkeyAuthHandler = createAuthMiddleware(unkey);
       server.addHook('preHandler', async (request, reply) => {
         if (request.url.startsWith('/api/v1/') && !request.billingPath) {
           await unkeyAuthHandler(request, reply);
         }
       });
       ```

       IMPORTANT: The Unkey auth hook now checks `!request.billingPath` before calling Unkey. If x402 already set it, Unkey is skipped entirely. If no x402 header was present, billingPath is still undefined so Unkey runs as before.

       OPTIMIZATION: Creating the handler on every request is wasteful. Better pattern: create the handler once outside the hook if env vars are available:
       ```typescript
       const walletAddress = process.env.RECEIVING_WALLET_ADDRESS;
       const x402PricePerCall = process.env.X402_PRICE_PER_CALL || '$0.001';
       const x402Network = process.env.X402_NETWORK || 'eip155:84532';

       server.addHook('preHandler', async (request, reply) => {
         if (x402Server && walletAddress && request.url.startsWith('/api/v1/')) {
           const handler = createX402PaymentHook(x402Server, walletAddress, x402PricePerCall, x402Network);
           await handler(request, reply);
         }
       });
       ```
       Even better: create the handler once at init time. But since x402Server is initialized asynchronously in start(), the simplest correct approach is to check `x402Server` on each request (it will be set after init).

    4. Add x402 settlement onResponse hook. Place it AFTER the existing Stripe meter-usage onResponse hook:

       ```typescript
           // ... rest of start()
       ```
